## 题解1

```java
/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode detectCycle(ListNode head) {
        ListNode l = head;
        ListNode r = head;

        while (r != null) {
            r = r.next;
            l = l.next;
            if (r != null) {
                r = r.next;
            }

            if (l == r) {
                break;
            }
        }
        if (r == null) {
            return null;
        }
        l = head;
        while (l != r) {
            l = l.next;
            r = r.next;
        }
        return l;
    }
}
```
