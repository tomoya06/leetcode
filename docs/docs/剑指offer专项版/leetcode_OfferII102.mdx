class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length;
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }
        int neg2 = sum-target;
        if (neg2 < 0 || neg2%2 != 0) {
            return 0;
        }
        int neg = neg2 / 2;

        int[][] dp = new int[n+1][neg+1];
        
        dp[0][0] = 1;

        for (int i=1; i<=n; i++) {
            int num = nums[i-1];
            for (int j=0; j<=neg; j++) {
                if (num > j) {
                    dp[i][j] = dp[i-1][j];
                } else {
                    dp[i][j] = dp[i-1][j] + dp[i-1][j-num];
                }
            }
        }

        return dp[n][neg];
    }
}

class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int sum = 0;
        for (int num : nums) {
            sum += num;
        }

        int neg2 = sum - target;
        if (neg2 < 0 || neg2 % 2 != 0) {
            return 0;
        }

        int n = nums.length, neg = neg2/2;
        int[] dp = new int[neg+1];
        dp[0] = 1;
        for (int i=0; i<n; i++) {
            int num = nums[i];
            for (int j=neg; j>=num; j--) {
                dp[j] += dp[j-num];
            }
        }

        return dp[neg];
    }
}
